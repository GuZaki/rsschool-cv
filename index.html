            add(productName, productPrice, productQuantity = 1) {
                const productObj = {
                    productName,
                    productPrice,
                    productQuantity,
                };
                this.items.push(productObj);
                this.increaseCount();
                this.calculateItemPrice();
            },

            increaseCount(num) {
                this.count += num;
            },

            calculateItemPrice() {
                return this.items.reduce((prev, item) => 
                prev + item.productPrice * item.productQuantity, 0);
            },

            clear() {
                this.items = [];
                this.totalPrice = 0;
                this.count = 0;
            },
